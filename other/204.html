#配置全局代理上网
adb shell settings put global http_proxy 192.168.1.2:8888
adb shell settings put global https_proxy 192.168.1.2:8888
adb shell settings put global socks4 192.168.1.2:8888
adb shell settings put global socks5 192.168.1.2:8888
adb shell settings put global all_proxy 192.168.1.2:8888
#查看代理上网配置
adb shell settings get global http_proxy
#取消代理上网
adb shell settings put global http_proxy :0
